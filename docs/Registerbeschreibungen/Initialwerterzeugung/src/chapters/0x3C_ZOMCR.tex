\subsection{Zoom Control Register (ZOMCR)}

Address offset: \textbf{0x3C}
\newline
Reset value: 0x0000 0001

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader[lsb=16]{16-31} \\
        \bitbox{16}{Res}
    \end{bytefield}
\end{figure}

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader{0-15} \\
        \bitbox{8}{Res} &
        \bitbox{8}{DS} \\
        \bitbox[]{8}{}
        \bitbox[]{8}{R/W}
    \end{bytefield}
\end{figure}

\begin{itemize}
    \item \textbf{Bit 31:8 - Reserved Bits}
    \item \textbf{Bit 7:0 - DS: Distance}\\
    Faktor für den Koordinatenabstand zwischen zwei Pixeln.
\end{itemize}
