\subsection{XOR Mask IM Control Register (XMICR)}

Address offset: \textbf{0x28}
\newline
Reset value: 0x0000 0001

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader[lsb=16]{16-31} \\
        \bitbox{16}{Res}
    \end{bytefield}
\end{figure}

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader{0-15} \\
        \bitbox{16}{XI} \\
        \bitbox[]{16}{R/W}
    \end{bytefield}
\end{figure}

\begin{itemize}
    \item \textbf{Bit 31:16 - Reserved Bits}
    \item \textbf{Bit 15:0 - XI: XOR IM}\\
    Maske der XOR für das LFSR des IM Animationsziels.
    (1: XOR, 0: kein XOR)
\end{itemize}
