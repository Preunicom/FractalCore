\subsection{Diamond Height Control Register (DHCR)}

Address offset: \textbf{0x38}
\newline
Reset value: 0x0000 00FF

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader[lsb=16]{16-31} \\
        \bitbox{16}{Res}
    \end{bytefield}
\end{figure}

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader{0-15} \\
        \bitbox{16}{DH} \\
        \bitbox[]{16}{R/W}
    \end{bytefield}
\end{figure}

\begin{itemize}
    \item \textbf{Bit 31:16 - Reserved Bits}
    \item \textbf{Bit 15:0 - DH: Diamond Height}\\
    Faktor für die Höhe der Raute im Diamond Mode.
    \item (IM Anteil)
\end{itemize}
