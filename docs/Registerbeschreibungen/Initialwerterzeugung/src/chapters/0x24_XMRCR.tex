\subsection{XOR Mask RE Control Register (XMRCR)}

Address offset: \textbf{0x24}
\newline
Reset value: 0x0000 0001

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader[lsb=16]{16-31} \\
        \bitbox{16}{Res}
    \end{bytefield}
\end{figure}

\begin{figure}[H]
    \centering
    \begin{bytefield}[bitwidth=\bflenhalfword, endianness=big]{16}
        \bitheader{0-15} \\
        \bitbox{16}{XR} \\
        \bitbox[]{16}{R/W}
    \end{bytefield}
\end{figure}

\begin{itemize}
    \item \textbf{Bit 31:16 - Reserved Bits}
    \item \textbf{Bit 15:0 - XR: XOR RE}\\
    Maske der XOR für das LFSR des RE Animationsziels.
    (1: XOR, 0: kein XOR)
\end{itemize}
